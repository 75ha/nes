想象下我们如何实现取得 多少道多少的元素 {a,b} 大于等于nth a 小于nth b的元素集合 在进行语法扩展时，name不是必须的，如下面的range 但是为了维护理解方便,建议还是加上，reg是必须的， action与filter至少需要需要二选其一(根据场景不同)才能完成一个自定义规则的实施

 
nes.addRule("range",{
  reg:/\s*\{\s*(\d*),(\-?\d*)\s*\}\s*/,

action中需要注意两个部分一个是this.error()打印出解析错误信息， 一个this.current()永远返回当前的Simple Selector对应的data部分这是个hash表， 你往里放key value对就行

 
  action:function(all, a, b){

    var current = this.current(), //this.current返回当前匹配的simple selector
      pesudos = current.pesudos || (current.pesudos = []) 

    a = a && parseInt(a) || 1
    b = b && parseInt(b) || 0
    pesudos.push({  //a 如果不存在 视为
      name:"nth-child",
      param:{
        start:a,
        step:1 
      }
    })  
    if(b>0){
      pesudos.push({ //意思小于b
        name:"not",
        param:":nth-child(" + (b+1) + ")"
      })
    }else{
      pesudos.push({  
        name:"nth-last-child",
        param:{
          start:b?-b:1,
          step:1
        }
      })
    }
  }
})
